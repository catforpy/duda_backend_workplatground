### 第二阶段：创建新表（6张表）

#### 1. 用户扩展资料表

```sql
CREATE TABLE user_profiles (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    nickname VARCHAR(50) COMMENT '昵称',
    gender TINYINT COMMENT '性别：0-未知, 1-男, 2-女',
    birthday DATE COMMENT '生日',
    id_card VARCHAR(18) COMMENT '身份证号',
    bio VARCHAR(500) COMMENT '个人简介',
    signature VARCHAR(200) COMMENT '个性签名',
    qq VARCHAR(20) COMMENT 'QQ号',
    wechat VARCHAR(50) COMMENT '微信号',
    tags VARCHAR(500) COMMENT '兴趣标签（逗号分隔）',
    privacy_level TINYINT DEFAULT 1 COMMENT '隐私级别：1-公开, 2-仅好友, 3-私密',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    UNIQUE KEY uk_user_id (user_id),
    INDEX idx_tags (tags)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户扩展资料表';
```

#### 2. 用户地址表

```sql
CREATE TABLE user_addresses (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    contact_name VARCHAR(50) COMMENT '联系人姓名',
    contact_phone VARCHAR(20) COMMENT '联系人电话',
    province VARCHAR(50) COMMENT '省份',
    province_code VARCHAR(20) COMMENT '省份代码',
    city VARCHAR(50) COMMENT '城市',
    city_code VARCHAR(20) COMMENT '城市代码',
    district VARCHAR(50) COMMENT '区县',
    district_code VARCHAR(20) COMMENT '区县代码',
    detail_address VARCHAR(200) COMMENT '详细地址',
    postal_code VARCHAR(10) COMMENT '邮政编码',
    tag VARCHAR(20) COMMENT '地址标签：home-家, company-公司',
    is_default TINYINT DEFAULT 0 COMMENT '是否默认地址：1-是, 0-否',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_user_id (user_id),
    INDEX idx_is_default (is_default)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户地址表';
```

#### 3. 第三方账号绑定表

```sql
CREATE TABLE user_third_parties (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    platform VARCHAR(20) NOT NULL COMMENT '第三方平台：wechat, qq, alipay, weibo',
    openid VARCHAR(100) NOT NULL COMMENT '第三方OpenID',
    unionid VARCHAR(100) COMMENT '第三方UnionID',
    nickname VARCHAR(100) COMMENT '第三方昵称',
    avatar_url VARCHAR(500) COMMENT '第三方头像',
    bind_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '绑定时间',
    unbind_time DATETIME COMMENT '解绑时间',
    status TINYINT DEFAULT 1 COMMENT '状态：1-已绑定, 0-已解绑',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    UNIQUE KEY uk_platform_openid (platform, openid),
    INDEX idx_user_id (user_id),
    INDEX idx_unionid (unionid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='第三方账号绑定表';
```

#### 4. 用户认证日志表

```sql
CREATE TABLE user_auth_logs (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键ID',
    user_id BIGINT COMMENT '用户ID',
    auth_type VARCHAR(20) NOT NULL COMMENT '认证类型：register, login, logout, refresh',
    login_type VARCHAR(20) COMMENT '登录方式：phone_password, email_password, third_party',
    client_type VARCHAR(20) COMMENT '客户端类型：web, app, mini_program',
    client_ip VARCHAR(50) COMMENT '客户端IP',
    device_id VARCHAR(100) COMMENT '设备唯一标识',
    user_agent VARCHAR(500) COMMENT 'User-Agent',
    country VARCHAR(50) COMMENT '国家',
    province VARCHAR(50) COMMENT '省份',
    city VARCHAR(50) COMMENT '城市',
    status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：1-成功, 0-失败',
    fail_reason VARCHAR(200) COMMENT '失败原因',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    INDEX idx_user_id (user_id),
    INDEX idx_auth_type (auth_type),
    INDEX idx_client_ip (client_ip),
    INDEX idx_create_time (create_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户认证日志表';
```

#### 5. 用户角色表（可选）

```yaml
mode:
  type: Standalone
  repository:
    type: JDBC

  master:
    dataSourceClassName: com.alibaba.druid.pool.DruidDataSource
    driverClassName: com.mysql.cj.jdbc.Driver
    url: jdbc:mysql://120.26.170.213:3306/duda_nexus?useSSL=false&serverTimezone=Asia/Shanghai
    username: root
    password: duda2024
    initialSize: 5
    minIdle: 5
    maxActive: 20

  slave:
    dataSourceClassName: com.alibaba.druid.pool.DruidDataSource
    driverClassName: com.mysql.cj.jdbc.Driver
    url: jdbc:mysql://120.26.170.213:3306/duda_nexus?useSSL=false&serverTimezone=Asia/Shanghai
    username: root
    password: duda2024
    initialSize: 5
    minIdle: 5
    maxActive: 20

rules:
  # 读写分离
  - !READWRITE_SPLITTING
    dataSources:
      readwrite_ds:
        writeDataSourceName: master
        readDataSourceNames:
          - slave
        loadBalancerName: round_robin
    loadBalancers:
      round_robin:
        type: ROUND_ROBIN

  # 分表规则
  - !SHARDING
    tables:
      # users表分100张
      users:
        actualDataNodes: readwrite_ds.users_$->{['00'..'99']}
        tableStrategy:
          standard:
            shardingColumn: id
            shardingAlgorithmName: users_mod

      # user_profiles表跟随users分片
      user_profiles:
        actualDataNodes: readwrite_ds.user_profiles_$->{['00'..'99']}
        tableStrategy:
          standard:
            shardingColumn: user_id
            shardingAlgorithmName: user_profiles_mod

    shardingAlgorithms:
      users_mod:
        type: MOD
        props:
          sharding-count: 100

      user_profiles_mod:
        type: MOD
        props:
          sharding-count: 100

props:
  sql-show: true
```

## 📊 最终架构

```
┌──────────────────────────────────────────────┐
│            前端 / API 调用者                  │
└─────────────────┬────────────────────────────┘
                  │
                  ↓
┌──────────────────────────────────────────────┐
│           AuthController                    │
│     (对外API接口 - 完全不变)                   │
│  /api/auth/register, /api/auth/login         │
└─────────────────┬────────────────────────────┘
                  │
                  ↓
┌──────────────────────────────────────────────┐
│            AuthService                       │
│     (业务逻辑层 - 内部调用多个Service)         │
└─────────────────┬────────────────────────────┘
                  │
        ┌─────────┼─────────┬─────────┐
        ↓         ↓         ↓         ↓
   ┌────────┐ ┌──────────┐ ┌────────┐ ┌──────────┐
   │UserService││UserProfile││UserAddress││ThirdParty│
   │ (核心)  ││Service   ││Service   ││Service   │
   └────┬────┘ └─────┬────┘ └────┬────┘ └─────┬────┘
        │           │           │           │
        ↓           ↓           ↓           ↓
   ┌────────┐ ┌──────────┐ ┌────────┐ ┌──────────┐
   │ users  │ │user_     │ │user_    │ │user_     │
   │ (20字段)│ │profiles  │ │addresses│ │third_    │
   └────────┘ └──────────┘ └────────┘ └──────────┘
```
