#### 1.5 OpenFeign 接口

```java
package com.duda.user.api.feign;

import com.duda.common.core.web.Result;
import com.duda.user.api.dto.UserDTO;
import org.springframework.cloud.openfeign.FeignClient;
import org.springframework.web.bind.annotation.*;

/**
 * 用户服务 Feign 客户端
 * 被其他服务调用，实现服务间通信
 */
@FeignClient(
    name = "duda-user-provider",
    path = "/user"
)
public interface UserClient {

    /**
     * 根据用户ID查询用户
     *
     * @param userId 用户ID
     * @return 用户信息
     */
    @GetMapping("/{userId}")
    Result<UserDTO> getUserById(@PathVariable("userId") Long userId);

    /**
     * 创建用户
     *
     * @param userDTO 用户信息
     * @return 用户ID
     */
    @PostMapping
    Result<Long> createUser(@RequestBody UserDTO userDTO);

    /**
     * 更新用户
     *
     * @param userId  用户ID
     * @param userDTO 用户信息
     * @return 是否成功
     */
    @PutMapping("/{userId}")
    Result<Boolean> updateUser(
        @PathVariable("userId") Long userId,
        @RequestBody UserDTO userDTO
    );

    /**
     * 删除用户
     *
     * @param userId 用户ID
     * @return 是否成功
     */
    @DeleteMapping("/{userId}")
    Result<Boolean> deleteUser(@PathVariable("userId") Long userId);
}
```

#### 2.5 Service 实现类

```java
package com.duda.user.service.impl;

import com.duda.user.api.dto.UserDTO;
import com.duda.user.api.constants.UserConstants;
import com.duda.user.entity.UserPO;
import com.duda.user.mapper.UserMapper;
import com.duda.user.service.IUserService;
import com.baomidou.mybatisplus.core.conditions.query.QueryWrapper;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.BeanUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;

/**
 * 用户服务实现
 */
@Slf4j
@Service
public class UserServiceImpl implements IUserService {

    @Autowired
    private UserMapper userMapper;

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    @Override
    public UserDTO getById(Long userId) {
        // 1. 先从缓存获取
        String cacheKey = UserConstants.USER_CACHE_PREFIX + userId;
        UserDTO cachedUser = (UserDTO) redisTemplate.opsForValue().get(cacheKey);
        if (cachedUser != null) {
            log.info("从缓存获取用户: userId={}", userId);
            return cachedUser;
        }

        // 2. 从数据库查询
        UserPO userPO = userMapper.selectById(userId);
        if (userPO == null) {
            return null;
        }

        // 3. 转换为 DTO
        UserDTO userDTO = convertToDTO(userPO);

        // 4. 放入缓存
        redisTemplate.opsForValue().set(
            cacheKey,
            userDTO,
            UserConstants.USER_CACHE_EXPIRE,
            TimeUnit.SECONDS
        );

        return userDTO;
    }

    @Override
    public Long create(UserDTO userDTO) {
        UserPO userPO = convertToPO(userDTO);
        userPO.setCreateTime(LocalDateTime.now());
        userMapper.insert(userPO);
        return userPO.getId();
    }

    @Override
    public Boolean update(Long userId, UserDTO userDTO) {
        UserPO userPO = convertToPO(userDTO);
        userPO.setId(userId);
        userPO.setUpdateTime(LocalDateTime.now());
        int result = userMapper.updateById(userPO);

        // 清除缓存
        String cacheKey = UserConstants.USER_CACHE_PREFIX + userId;
        redisTemplate.delete(cacheKey);

        return result > 0;
    }

    @Override
    public Boolean delete(Long userId) {
        int result = userMapper.deleteById(userId);

        // 清除缓存
        String cacheKey = UserConstants.USER_CACHE_PREFIX + userId;
        redisTemplate.delete(cacheKey);

        return result > 0;
    }

    // ==================== 私有方法 ====================

    private UserDTO convertToDTO(UserPO userPO) {
        UserDTO userDTO = new UserDTO();
        BeanUtils.copyProperties(userPO, userDTO);
        return userDTO;
    }

    private UserPO convertToPO(UserDTO userDTO) {
        UserPO userPO = new UserPO();
        BeanUtils.copyProperties(userDTO, userPO);
        return userPO;
    }
}
```

#### 2.6 Controller

```java
package com.duda.user.controller;

import com.duda.common.core.web.Result;
import com.duda.user.api.dto.UserDTO;
import com.duda.user.service.IUserService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

/**
 * 用户控制器
 */
@Slf4j
@RestController
@RequestMapping("/user")
public class UserController {

    @Autowired
    private IUserService userService;

    /**
     * 根据用户ID查询用户
     */
    @GetMapping("/{userId}")
    public Result<UserDTO> getUserById(@PathVariable Long userId) {
        log.info("查询用户: userId={}", userId);
        UserDTO user = userService.getById(userId);
        return Result.success(user);
    }

    /**
     * 创建用户
     */
    @PostMapping
    public Result<Long> createUser(@RequestBody UserDTO userDTO) {
        log.info("创建用户: userDTO={}", userDTO);
        Long userId = userService.create(userDTO);
        return Result.success(userId);
    }

    /**
     * 更新用户
     */
    @PutMapping("/{userId}")
    public Result<Boolean> updateUser(
        @PathVariable Long userId,
        @RequestBody UserDTO userDTO
    ) {
        log.info("更新用户: userId={}, userDTO={}", userId, userDTO);
        Boolean result = userService.update(userId, userDTO);
        return Result.success(result);
    }

    /**
     * 删除用户
     */
    @DeleteMapping("/{userId}")
    public Result<Boolean> deleteUser(@PathVariable Long userId) {
        log.info("删除用户: userId={}", userId);
        Boolean result = userService.delete(userId);
        return Result.success(result);
    }
}
```

## 📝 开发规范

### 1. 接口模块开发规范

**包含内容：**
- ✅ DTO（数据传输对象）
- ✅ 常量（Constants）
- ✅ 枚举（Enums）
- ✅ OpenFeign 接口
- ✅ 视图对象（VO，可选）

**不包含内容：**
- ❌ 业务逻辑
- ❌ 数据库访问
- ❌ 第三方依赖（除 Lombok、OpenFeign、Validation）
- ❌ 实体类（PO）

### 2. 实现模块开发规范

**包含内容：**
- ✅ Controller（REST API）
- ✅ Service（业务逻辑）
- ✅ Mapper（数据库访问）
- ✅ Entity（数据库实体）
- ✅ Config（配置类）

**依赖关系：**
```xml
<dependency>
    <groupId>com.duda</groupId>
    <artifactId>duda-{业务名}-interface</artifactId>
</dependency>
```

### 3. 导入规范

**在实现模块中导入接口模块的内容：**

```java
// ✅ 正确：从 api 包导入
import com.duda.user.api.dto.UserDTO;
import com.duda.user.api.constants.UserConstants;
import com.duda.user.api.feign.UserClient;

// ❌ 错误：不要在实现模块中创建重复的 DTO
```
