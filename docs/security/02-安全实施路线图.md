##### 1.1 创建日志脱敏工具类

**文件路径**: `duda-common/duda-common-security/src/main/java/com/duda/common/security/util/DesensitizedUtil.java`

```java
package com.duda.common.security.util;

import java.util.regex.Pattern;

/**
 * 日志脱敏工具类
 *
 * 功能：
 * 1. 手机号脱敏：138****1234
 * 2. 身份证号脱敏：110123********1234
 * 3. 银行卡号脱敏：6222 **** **** 1234
 * 4. 邮箱脱敏：a***@example.com
 * 5. 密码字段：不输出日志
 */
public class DesensitizedUtil {

    // 手机号正则
    private static final Pattern PHONE_PATTERN = Pattern.compile("(\\d{3})\\d{4}(\\d{4})");

    // 身份证号正则
    private static final Pattern ID_CARD_PATTERN = Pattern.compile("(\\d{6})\\d{8}(\\d{4})");

    // 银行卡号正则
    private static final Pattern BANK_CARD_PATTERN = Pattern.compile("(\\d{4})\\d{8,12}(\\d{4})");

    // 邮箱正则
    private static final Pattern EMAIL_PATTERN = Pattern.compile("(\\w{1})\\w+(@\\w+\\.\\w+)");

    /**
     * 手机号脱敏
     * @param phone 原始手机号
     * @return 脱敏后的手机号：138****1234
     */
    public static String desensitizedPhone(String phone) {
        if (phone == null || phone.length() != 11) {
            return phone;
        }
        return PHONE_PATTERN.matcher(phone).replaceAll("$1****$2");
    }

    /**
     * 身份证号脱敏
     * @param idCard 原始身份证号
     * @return 脱敏后的身份证号：110123********1234
     */
    public static String desensitizedIdCard(String idCard) {
        if (idCard == null || idCard.length() != 18) {
            return idCard;
        }
        return ID_CARD_PATTERN.matcher(idCard).replaceAll("$1********$2");
    }

    /**
     * 银行卡号脱敏
     * @param bankCard 原始银行卡号
     * @return 脱敏后的银行卡号：6222 **** **** 1234
     */
    public static String desensitizedBankCard(String bankCard) {
        if (bankCard == null || bankCard.length() < 16) {
            return bankCard;
        }
        return BANK_CARD_PATTERN.matcher(bankCard).replaceAll("$1 **** **** $2");
    }

    /**
     * 邮箱脱敏
     * @param email 原始邮箱
     * @return 脱敏后的邮箱：a***@example.com
     */
    public static String desensitizedEmail(String email) {
        if (email == null || !email.contains("@")) {
            return email;
        }
        return EMAIL_PATTERN.matcher(email).replaceAll("$1***$2");
    }

    /**
     * 通用脱敏方法（自动识别类型）
     * @param content 原始内容
     * @return 脱敏后的内容
     */
    public static String desensitized(String content) {
        if (content == null) {
            return null;
        }

        // 递归处理（防止多次脱敏）
        String result = content;
        result = desensitizedPhoneInText(result);
        result = desensitizedIdCardInText(result);
        result = desensitizedBankCardInText(result);
        result = desensitizedEmailInText(result);

        return result;
    }

    /**
     * 脱敏文本中的所有手机号
     */
    private static String desensitizedPhoneInText(String text) {
        return PHONE_PATTERN.matcher(text).replaceAll("$1****$2");
    }

    /**
     * 脱敏文本中的所有身份证号
     */
    private static String desensitizedIdCardInText(String text) {
        return ID_CARD_PATTERN.matcher(text).replaceAll("$1********$2");
    }

    /**
     * 脱敏文本中的所有银行卡号
     */
    private static String desensitizedBankCardInText(String text) {
        return BANK_CARD_PATTERN.matcher(text).replaceAll("$1 **** **** $2");
    }

    /**
     * 脱敏文本中的所有邮箱
     */
    private static String desensitizedEmailInText(String text) {
        return EMAIL_PATTERN.matcher(text).replaceAll("$1***$2");
    }
}
```

##### 1.2 创建Logback自定义转换器

**文件路径**: `duda-common/duda-common-security/src/main/java/com/duda/common/security/log/DesensitizedConverter.java`

```java
package com.duda.common.security.log;

import ch.qos.logback.classic.pattern.MessageConverter;
import ch.qos.logback.classic.spi.ILoggingEvent;
import com.duda.common.security.util.DesensitizedUtil;

/**
 * Logback 脱敏转换器
 *
 * 使用方式：
 * 在 logback-spring.xml 中配置：
 * <conversionRule conversionWord="msg" converterClass="com.duda.common.security.log.DesensitizedConverter" />
 */
public class DesensitizedConverter extends MessageConverter {

    @Override
    public String convert(ILoggingEvent event) {
        String originalMsg = super.convert(event);

        // 对日志内容进行脱敏
        return DesensitizedUtil.desensitized(originalMsg);
    }
}
```

##### 1.3 配置Logback

**文件路径**: `src/main/resources/logback-spring.xml`

```xml
<?xml version="1.0" encoding="UTF-8"?>
<configuration>
    <!-- 引入脱敏转换器 -->
    <conversionRule conversionWord="msg" converterClass="com.duda.common.security.log.DesensitizedConverter"/>

    <!-- 控制台输出 -->
    <appender name="CONSOLE" class="ch.qos.logback.core.ConsoleAppender">
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{50} - %msg%n</pattern>
            <charset>UTF-8</charset>
        </encoder>
    </appender>

    <!-- 文件输出 -->
    <appender name="FILE" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <file>${LOG_PATH:-/var/log/duda}/application.log</file>
        <encoder>
            <pattern>%d{yyyy-MM-dd HH:mm:ss.SSS} [%thread] %-5level %logger{50} - %msg%n</pattern>
            <charset>UTF-8</charset>
        </encoder>
        <rollingPolicy class="ch.qos.logback.core.rolling.TimeBasedRollingPolicy">
            <fileNamePattern>${LOG_PATH:-/var/log/duda}/application-%d{yyyy-MM-dd}.log</fileNamePattern>
            <maxHistory>30</maxHistory>
        </rollingPolicy>
    </appender>

    <!-- 根日志配置 -->
    <root level="INFO">
        <appender-ref ref="CONSOLE"/>
        <appender-ref ref="FILE"/>
    </root>
</configuration>
```

##### 2.3 创建限流异常处理器

**文件路径**: `duda-common/duda-common-web/src/main/java/com/duda/common/web/sentinel/SentinelBlockExceptionHandler.java`

```java
package com.duda.common.web.sentinel;

import com.alibaba.csp.sentinel.adapter.spring.webmvc.callback.BlockExceptionHandler;
import com.alibaba.csp.sentinel.slots.block.BlockException;
import com.alibaba.csp.sentinel.slots.block.authority.AuthorityException;
import com.alibaba.csp.sentinel.slots.block.degrade.DegradeException;
import com.alibaba.csp.sentinel.slots.block.flow.FlowException;
import com.duda.common.web.model.Result;
import com.duda.common.web.enums.ErrorCode;
import com.fasterxml.jackson.databind.ObjectMapper;
import org.springframework.stereotype.Component;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * Sentinel 限流异常处理器
 */
@Component
public class SentinelBlockExceptionHandler implements BlockExceptionHandler {

    private final ObjectMapper objectMapper = new ObjectMapper();

    @Override
    public void handle(HttpServletRequest request,
                       HttpServletResponse response,
                       BlockException e) throws Exception {
        response.setStatus(429);
        response.setContentType("application/json;charset=UTF-8");

        Result<?> result;

        if (e instanceof FlowException) {
            // 限流异常
            result = Result.error(ErrorCode.RATE_LIMIT_EXCEEDED, "请求过于频繁，请稍后再试");
        } else if (e instanceof DegradeException) {
            // 熔断异常
            result = Result.error(ErrorCode.SERVICE_DEGRADED, "服务暂时不可用，请稍后再试");
        } else if (e instanceof AuthorityException) {
            // 授权异常
            result = Result.error(ErrorCode.UNAUTHORIZED, "无权访问");
        } else {
            // 其他异常
            result = Result.error(ErrorCode.SERVER_ERROR, "服务器繁忙");
        }

        response.getWriter().write(objectMapper.writeValueAsString(result));
    }
}
```

public class SentinelRateLimitTest {

        for (int i = 0; i < 11; i++) {
            MvcResult result = mockMvc.perform(post("/api/auth/v2/platform-admin/login/password")

                    .andReturn();

                // 前10次应该成功

                // 第11次应该被限流
                assertEquals(429, result.getResponse().getStatus());
            }
        }
    }

- [ ] 安全审计：记录用户操作日志，留存6个月以上
- [ ] 数据完整性：检测重要数据完整性
- [ **] 数据保密性：采用密码技术保证敏感数据存储保密性
- [ ] 数据备份恢复：每天备份，异地备份
- [ ] 客户端防恶意代码：安装防病毒软件

### 等保三级要求

- [ ] 所有二级要求
- [ ] 密钥管理：建立密钥管理制度
- [ ] 传输加密：所有数据传输加密
- [ ] 入侵防范：部署入侵检测系统
- [ ] 恶意代码防范：建立恶意代码防范机制
- [ ] 安全审计：审计记录定期备份

### 个人信息保护法合规

- [ ] 明确告知用户个人信息处理范围
- [ ] 获取用户明确同意
- [ ] 敏感信息单独同意
- [ ] 提供个人信息查询、更正、删除通道
- [ ] 15个工作日内响应删除请求
- [ ] 数据泄露72小时内告知用户

## 🎓 最佳实践

### 1. 安全左移
在开发阶段就考虑安全，而不是后期补丁。

### 2. 最小权限原则
默认拒绝，按需授权。

### 3. 纵深防御
多层防护，单一防护失效不影响整体安全。

### 4. 持续审计
定期审计安全配置和日志。

### 5. 安全培训
定期对开发人员进行安全培训。
