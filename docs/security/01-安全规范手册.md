### 1.2 JWT认证实现

#### 1.2.1 JWT工具类

```java
@Component
@Slf4j
public class JwtUtils {

    @Value("${duda.auth.jwt.secret}")
    private String secret;

    @Value("${duda.auth.jwt.expire}")
    private Long expire;

    @Value("${duda.auth.jwt.refresh-expire}")
    private Long refreshExpire;

    /**
     * 生成访问Token
     */
    public String generateToken(Long userId, String userName) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("userId", userId);
        claims.put("userName", userName);

        return Jwts.builder()
                .setClaims(claims)
                .setSubject(userName)
                .setIssuedAt(new Date())
                .setExpiration(new Date(System.currentTimeMillis() + expire * 1000))
                .signWith(SignatureAlgorithm.HS256, secret)
                .compact();
    }

    /**
     * 生成刷新Token
     */
    public String generateRefreshToken(Long userId) {
        return Jwts.builder()
                .setSubject(String.valueOf(userId))
                .setIssuedAt(new Date())
                .setExpiration(new Date(System.currentTimeMillis() + refreshExpire * 1000))
                .signWith(SignatureAlgorithm.HS256, secret)
                .compact();
    }

    /**
     * 解析Token
     */
    public Claims parseToken(String token) {
        try {
            return Jwts.parser()
                    .setSigningKey(secret)
                    .parseClaimsJws(token)
                    .getBody();
        } catch (Exception e) {
            log.error("解析Token失败", e);
            return null;
        }
    }

    /**
     * 验证Token
     */
    public boolean validateToken(String token) {
        Claims claims = parseToken(token);
        if (claims == null) {
            return false;
        }

        Date expiration = claims.getExpiration();
        return expiration.after(new Date());
    }

    /**
     * 获取用户ID
     */
    public Long getUserId(String token) {
        Claims claims = parseToken(token);
        if (claims != null) {
            return claims.get("userId", Long.class);
        }
        return null;
    }
}
```

#### 1.2.2 认证服务

```java
@Service
@Slf4j
public class AuthServiceImpl implements AuthService {

    @Autowired
    private JwtUtils jwtUtils;

    @Autowired
    private UserClient userClient;

    @Autowired
    private RedisUtils redisUtils;

    /**
     * 用户登录
     */
    @Override
    public LoginVO login(LoginCommand cmd) {
        // 1. 验证用户名密码
        UserVO user = userClient.validateUser(cmd.getUserName(), cmd.getPassword());
        if (user == null) {
            throw new BusinessException("用户名或密码错误");
        }

        // 2. 生成Token
        String token = jwtUtils.generateToken(user.getId(), user.getUserName());
        String refreshToken = jwtUtils.generateRefreshToken(user.getId());

        // 3. 存储Token到Redis
        String tokenKey = RedisKey.getUserTokenKey(user.getId());
        redisUtils.set(tokenKey, token, 1, TimeUnit.HOURS);

        String refreshKey = RedisKey.getUserRefreshTokenKey(user.getId());
        redisUtils.set(refreshKey, refreshToken, 7, TimeUnit.DAYS);

        // 4. 返回登录信息
        return LoginVO.builder()
                .userId(user.getId())
                .userName(user.getUserName())
                .token(token)
                .refreshToken(refreshToken)
                .expireIn(3600)
                .build();
    }

    /**
     * 刷新Token
     */
    @Override
    public String refreshToken(String refreshToken) {
        // 1. 验证RefreshToken
        Claims claims = jwtUtils.parseToken(refreshToken);
        if (claims == null) {
            throw new BusinessException("刷新Token无效");
        }

        Long userId = Long.parseLong(claims.getSubject());

        // 2. 检查Redis中的RefreshToken
        String refreshKey = RedisKey.getUserRefreshTokenKey(userId);
        String savedToken = redisUtils.get(refreshKey);
        if (!refreshToken.equals(savedToken)) {
            throw new BusinessException("刷新Token已失效");
        }

        // 3. 生成新Token
        UserVO user = userClient.getUserById(userId);
        String newToken = jwtUtils.generateToken(user.getId(), user.getUserName());

        // 4. 更新Redis
        String tokenKey = RedisKey.getUserTokenKey(userId);
        redisUtils.set(tokenKey, newToken, 1, TimeUnit.HOURS);

        return newToken;
    }

    /**
     * 用户登出
     */
    @Override
    public void logout(Long userId) {
        // 删除Token
        String tokenKey = RedisKey.getUserTokenKey(userId);
        redisUtils.delete(tokenKey);

        // 删除RefreshToken
        String refreshKey = RedisKey.getUserRefreshTokenKey(userId);
        redisUtils.delete(refreshKey);
    }
}
```

```java
@Aspect
@Component
@Slf4j
public class PermissionAspect {

    @Autowired
    private PermissionService permissionService;

    @Around("@annotation(requirePermission)")
    public Object checkPermission(ProceedingJoinPoint point, RequirePermission requirePermission) throws Throwable {
        // 1. 获取当前用户
        Long userId = SecurityContext.getUserId();
        if (userId == null) {
            throw new PermissionException("用户未登录");
        }

        // 2. 检查权限
        String permissionCode = requirePermission.value();
        boolean hasPermission = permissionService.hasPermission(userId, permissionCode);

        if (!hasPermission) {
            log.warn("用户无权限: userId={}, permission={}", userId, permissionCode);
            throw new PermissionException("无操作权限");
        }

        // 3. 执行方法
        return point.proceed();
    }
}
```

### 2.2 敏感数据加密

```java
@Component
public class AesUtils {

    @Value("${duda.security.aes.key}")
    private String key;

    private static final String ALGORITHM = "AES/CBC/PKCS5Padding";

    /**
     * 加密
     */
    public String encrypt(String content) {
        try {
            Cipher cipher = Cipher.getInstance(ALGORITHM);
            SecretKeySpec keySpec = new SecretKeySpec(key.getBytes(), "AES");
            IvParameterSpec ivSpec = new IvParameterSpec(key.getBytes(), 0, 16);
            cipher.init(Cipher.ENCRYPT_MODE, keySpec, ivSpec);

            byte[] encrypted = cipher.doFinal(content.getBytes());
            return Base64.getEncoder().encodeToString(encrypted);
        } catch (Exception e) {
            throw new RuntimeException("加密失败", e);
        }
    }

    /**
     * 解密
     */
    public String decrypt(String content) {
        try {
            Cipher cipher = Cipher.getInstance(ALGORITHM);
            SecretKeySpec keySpec = new SecretKeySpec(key.getBytes(), "AES");
            IvParameterSpec ivSpec = new IvParameterSpec(key.getBytes(), 0, 16);
            cipher.init(Cipher.DECRYPT_MODE, keySpec, ivSpec);

            byte[] decrypted = cipher.doFinal(Base64.getDecoder().decode(content));
            return new String(decrypted);
        } catch (Exception e) {
            throw new RuntimeException("解密失败", e);
        }
    }
}
```

### 2.3 数据脱敏

```java
public class DataMaskingUtils {

    /**
     * 手机号脱敏
     */
    public static String maskPhone(String phone) {
        if (StringUtils.isEmpty(phone) || phone.length() != 11) {
            return phone;
        }
        return phone.substring(0, 3) + "****" + phone.substring(7);
    }

    /**
     * 邮箱脱敏
     */
    public static String maskEmail(String email) {
        if (StringUtils.isEmpty(email)) {
            return email;
        }
        int index = email.indexOf("@");
        if (index <= 1) {
            return email;
        }
        return email.charAt(0) + "***" + email.substring(index - 1);
    }

    /**
     * 身份证号脱敏
     */
    public static String maskIdCard(String idCard) {
        if (StringUtils.isEmpty(idCard) || idCard.length() != 18) {
            return idCard;
        }
        return idCard.substring(0, 6) + "********" + idCard.substring(14);
    }

    /**
     * 银行卡号脱敏
     */
    public static String maskBankCard(String bankCard) {
        if (StringUtils.isEmpty(bankCard) || bankCard.length() < 8) {
            return bankCard;
        }
        return bankCard.substring(0, 4) + "********" + bankCard.substring(bankCard.length() - 4);
    }
}
```

## 3. 接口安全

### 3.1 接口签名验证

```java
@Component
@Slf4j
public class ApiSignatureUtils {

    @Value("${duda.security.api.app-id}")
    private String appId;

    @Value("${duda.security.api.app-secret}")
    private String appSecret;

    /**
     * 生成签名
     */
    public String generateSign(Map<String, String> params) {
        // 1. 参数排序
        TreeMap<String, String> sortedParams = new TreeMap<>(params);

        // 2. 拼接参数
        StringBuilder sb = new StringBuilder();
        for (Map.Entry<String, String> entry : sortedParams.entrySet()) {
            if (!"sign".equals(entry.getKey())) {
                sb.append(entry.getKey()).append("=").append(entry.getValue()).append("&");
            }
        }

        // 3. 添加密钥
        sb.append("appSecret=").append(appSecret);

        // 4. MD5加密
        return DigestUtils.md5Hex(sb.toString());
    }

    /**
     * 验证签名
     */
    public boolean verifySign(Map<String, String> params, String sign) {
        String calculatedSign = generateSign(params);
        return calculatedSign.equalsIgnoreCase(sign);
    }
}
```

### 3.2 防重放攻击

```java
@Component
@Slf4j
public class ReplayAttackUtils {

    @Autowired
    private RedisUtils redisUtils;

    private static final long NONCE_EXPIRE_TIME = 300; // 5分钟

    /**
     * 检查nonce
     */
    public void checkNonce(String nonce, long timestamp) {
        // 1. 检查时间戳（5分钟内有效）
        long now = System.currentTimeMillis() / 1000;
        if (Math.abs(now - timestamp) > NONCE_EXPIRE_TIME) {
            throw new SecurityException("请求已过期");
        }

        // 2. 检查nonce是否重复
        String nonceKey = RedisKey.getApiNonceKey(nonce);
        Boolean exists = redisUtils.hasKey(nonceKey);

        if (Boolean.TRUE.equals(exists)) {
            throw new SecurityException("重复请求");
        }

        // 3. 存储nonce
        redisUtils.set(nonceKey, "1", NONCE_EXPIRE_TIME, TimeUnit.SECONDS);
    }
}
```

### 3.3 限流防刷

```java
@Component
@Slf4j
public class RateLimitUtils {

    @Autowired
    private RedisUtils redisUtils;

    /**
     * 令牌桶限流
     */
    public boolean rateLimit(String key, int limit, int period) {
        String rateKey = RedisKey.getRateLimitKey(key);

        Long count = redisUtils.increment(rateKey, 1);
        if (count == 1) {
            redisUtils.expire(rateKey, period, TimeUnit.SECONDS);
        }

        return count <= limit;
    }

    /**
     * IP限流
     */
    public boolean ipRateLimit(String ip, int limit, int period) {
        String key = "rate:limit:ip:" + ip;
        return rateLimit(key, limit, period);
    }

    /**
     * 用户限流
     */
    public boolean userRateLimit(Long userId, int limit, int period) {
        String key = "rate:limit:user:" + userId;
        return rateLimit(key, limit, period);
    }
}
```

## 4. XSS防护

### 4.1 输入过滤

```java
@Component
public class XssFilter implements Filter {

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
            throws IOException, ServletException {
        XssHttpServletRequestWrapper xssRequest = new XssHttpServletRequestWrapper((HttpServletRequest) request);
        chain.doFilter(xssRequest, response);
    }

    @Override
    public void init(FilterConfig filterConfig) throws ServletException {
    }

    @Override
    public void destroy() {
    }
}

public class XssHttpServletRequestWrapper extends HttpServletRequestWrapper {

    public XssHttpServletRequestWrapper(HttpServletRequest request) {
        super(request);
    }

    @Override
    public String getParameter(String name) {
        String value = super.getParameter(name);
        return cleanXss(value);
    }

    @Override
    public String[] getParameterValues(String name) {
        String[] values = super.getParameterValues(name);
        if (values != null) {
            for (int i = 0; i < values.length; i++) {
                values[i] = cleanXss(values[i]);
            }
        }
        return values;
    }

    private String cleanXss(String value) {
        if (value == null) {
            return null;
        }

        // HTML转义
        value = StringEscapeUtils.escapeHtml4(value);

        // 移除危险字符
        value = value.replaceAll("<", "&lt;").replaceAll(">", "&gt;");
        value = value.replaceAll("\\(", "&#40;").replaceAll("\\)", "&#41;");
        value = value.replaceAll("'", "&#39;");
        value = value.replaceAll("eval\\((.*)\\)", "");
        value = value.replaceAll("[\\\"\\\'][\\s]*javascript:(.*)[\\\"\\\']", "\"\"");
        value = value.replaceAll("script", "");

        return value;
    }
}
```

```java
@Component
public class SqlInjectionUtils {

    private static final Pattern SQL_PATTERN = Pattern.compile(
            ".*(union|select|insert|update|delete|drop|truncate|exec|execute|script|javascript|alert).*",
            Pattern.CASE_INSENSITIVE
    );

    /**
     * 检查SQL注入
     */
    public static boolean checkSqlInjection(String value) {
        if (StringUtils.isEmpty(value)) {
            return false;
        }
        return SQL_PATTERN.matcher(value).matches();
    }

    /**
     * 过滤SQL关键字
     */
    public static String filterSql(String value) {
        if (StringUtils.isEmpty(value)) {
            return value;
        }
        return value.replaceAll("union|select|insert|update|delete|drop|truncate|exec|execute", "");
    }
}
```

```java
@Configuration
public class SecurityHeaderConfig {

    @Bean
    public FilterRegistrationBean<SecurityHeaderFilter> securityHeaderFilter() {
        FilterRegistrationBean<SecurityHeaderFilter> registration = new FilterRegistrationBean<>();

public class SecurityHeaderFilter implements Filter {

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
            throws IOException, ServletException {
        HttpServletResponse httpResponse = (HttpServletResponse) response;

        // 防止点击劫持
        httpResponse.setHeader("X-Frame-Options", "DENY");

        // 防止MIME类型混淆
        httpResponse.setHeader("X-Content-Type-Options", "nosniff");

        // 启用XSS保护
        httpResponse.setHeader("X-XSS-Protection", "1; mode=block");

        // 内容安全策略
        httpResponse.setHeader("Content-Security-Policy", "default-src 'self'");

        // Referrer策略
        httpResponse.setHeader("Referrer-Policy", "no-referrer");

        chain.doFilter(request, response);
    }
}
```

```java
@Aspect
@Component
@Slf4j
public class AuditLogAspect {

    @Autowired
    private AuditLogService auditLogService;

    @Around("@annotation(auditLog)")
    public Object log(ProceedingJoinPoint point, AuditLog auditLog) throws Throwable {
        // 1. 获取请求信息
        HttpServletRequest request = ((ServletRequestAttributes) RequestContextHolder.getRequestAttributes()).getRequest();
        Long userId = SecurityContext.getUserId();
        String operation = auditLog.value();
        String ip = IpUtils.getIpAddress(request);

        // 2. 执行方法
        long startTime = System.currentTimeMillis();
        Object result = null;
        try {
            result = point.proceed();
            long endTime = System.currentTimeMillis();
            long duration = endTime - startTime;

            // 3. 记录成功日志
            auditLogService.recordSuccess(userId, operation, ip, duration);
            return result;
        } catch (Exception e) {
            // 4. 记录失败日志
            auditLogService.recordFailure(userId, operation, ip, e.getMessage());
            throw e;
        }
    }
}
```

## 9. 安全最佳实践

1. **永远不要信任用户输入**
   - 所有输入必须验证和过滤
   - 使用白名单而不是黑名单

2. **最小权限原则**
   - 数据库使用最小权限账号
   - 应用程序只申请必要的权限

3. **定期更新依赖**
   - 及时修复安全漏洞
   - 使用依赖扫描工具

4. **敏感数据加密**
   - 传输加密（HTTPS）
   - 存储加密（敏感字段）

5. **实施代码审查**
   - 安全代码审查
   - 使用静态分析工具

6. **安全测试**
   - 渗透测试
   - 漏洞扫描

7. **监控告警**
   - 异常登录告警
   - 异常操作告警
   - 安全事件日志

8. **备份与恢复**
   - 定期备份
   - 灾难恢复演练
